// Traditional function expressions
function add(a, b) {
  return a + b;
}
 
function subtract(a, b) {
  return a - b;
}
 
function multiply(a, b) {
  return a * b;
}
 
function divide(a, b) {
  if (b !== 0) {
    return a / b;
  } else {
    return "Cannot divide by zero";
  }
}
 
function square(num) {
  return multiply(num, num);
}
 
function greet(name) {
  if (name) {
    return "Hello, " + name + "!";
  } else {
    return "Hello, stranger!";
  }
}

console.log(square(4)); // Output: 16
